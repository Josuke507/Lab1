void Sock_Merchant()
{
int n,i,j,c,count=0;
	system("cls");
printf("\nINGRESE LA CANTIDAD DE CALCETINES: ");
scanf("%d",&n);
int arr[n];
for(i=0;i<n;i++){
printf("\nINGRESE EL NUMERO DEL CALCETIN#%d: ",i+1);
scanf("%d",&arr[i]);
}
for(i=0;i<n;i++){
    c=1;
    if(arr[i]!=0){
        for(j=i+1;j<n;j++){
        if(arr[i]==arr[j]){
            c++;
            arr[j]=0;
        }
    }
    count+=c/2;
    }
}
printf("\nLOS PARES DE LOS CALCETINES SON: %d\n",count);
system("pause");	
}
